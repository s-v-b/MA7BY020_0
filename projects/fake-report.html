neat_color_scale <-
      c("Africa" = "#01d4e5",
        "Americas" = "#7dea01" ,
        "Asia" = "#fc5173",
        "Europe" = "#fde803",
        "Oceania" = "#536227")
a_year <- sample(gapminder$year, 1)

p <- gapminder |>
    filter(year==a_year) |>
    ggplot() +
    aes(x = gdpPercap) +
    aes(y = lifeExp) +
    aes(size = pop) +
    aes(text = country) +                   #
    aes(fill = continent) +
    aes(color= continent) +
    aes(frame = year) +                     #
    geom_point(alpha=.5) +
    scale_x_log10() +
    scale_size_area(max_size = 15,
                    labels= scales::label_number(scale=1/1e6,
                                                suffix=" M")) +
    scale_fill_manual(values = neat_color_scale) +
    labs(title= glue("Gapminder  {a_year}"),
        x = "Yearly Income per Capita",
        y = "Life Expectancy",
        caption="From sick  and poor (bottom left) to healthy and rich (top right)")
